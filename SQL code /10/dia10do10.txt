create table NOTA(

ID	int unsigned not null auto_increment,
DISCIPLINA varchar(20) not null,
AV1 float(4,2),
AV2 float(4,2),
MATRICULA char(5),
primary key(ID),
foreign key(MATRICULA) references ALUNO(MATRICULA)

);


describe NOTA;
